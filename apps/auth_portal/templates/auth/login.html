تهيئة الجداول try: from apps.models.admin_db import AdminUser db.create_all() except Exception as e: print(f"❌ [Database Error]: {e}") @login_manager.user_loader def load_user(user_id): return AdminUser.query.get(int(user_id)) if user_id else None #
